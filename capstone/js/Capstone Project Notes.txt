const fragment = document.createDocumentFragment();

products.forEach(product => {
  const elem = document.createElement('elem');
  elem.textContent = product.var1;

  const elem2 = document.createElement('elem2');
  elem2.textContent = product.var2;

  const elem10 = document.createElement('elem10');
  elem10.textContent = product.var10;

  fragment.appendChild(elem);
  fragment.appendChild(elem2);
  fragment.appendChild(elem10);
});

container.appendChild(fragment); // Append everything at once

  },
  body: JSON.stringify({ key1: 'value1', key2: 'value2' }) // Convert data to JSON string
})
